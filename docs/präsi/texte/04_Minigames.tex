\begin{frame}
	\frametitle{Ziele}
	\begin{itemize}
		\item Erhalten von Items, zur Nutzung im Kampf
		\item Abwechslung gegenüber den Kämpfen
	\end{itemize}
\end{frame}

\begin{frame}
	\frametitle{Allgemein}
	\begin{itemize}
		\item ein Minigame pro Fachschaft
		\item drei Level, progressiv schwerer
		\begin{itemize}
			\item Level 1: Einführung
			\item Level 2: \glqq standard\grqq-Schwierigkeit
			\item Level 3: Herausforderung
		\end{itemize}
		\item Item als Belohnung, der Schwierigkeit entsprechend
		\item jederzeit Spielbar
	\end{itemize}
\end{frame}

\begin{frame}
\frametitle{ChemieLevelScene}
Eine Erweiterung der \texttt{LevelScene}, welche das \texttt{ChemieGameObject} zur \texttt{objects}-Liste hinzufügt.

Sie beinhaltet:
	\begin{itemize}
		\item \texttt{reset()}
		\item \texttt{ChemieLevelScene(int level)}
	\end{itemize}
\end{frame}

\begin{frame}
	\frametitle{Reagenzglas}
	Eine Erweiterung von \texttt{BaseObject}, die ein einzelnes Reagenzglas darstellt.
	Die Klasse beinhaltet:
		\begin{itemize}
			\item \texttt{MAX\_KAPAZITAET}, \texttt{GLAS\_BREITE}, \texttt{GLAS\_HOEHE}: Statische Konstanten
			\item \texttt{blockBilder}: \texttt{Map<String, Image>}.
			\item \texttt{blockHinzufuegen(String farbe)}
			\item \texttt{oberstenBlockEntfernen()}
		\end{itemize}
\end{frame}

\begin{frame}
	\frametitle{Reagenzglas}

	\begin{itemize}
		\item \texttt{oberstenBlockAnschauen()}
		\item \texttt{istLeer()}, \texttt{istVoll()}
		\item \texttt{istGeloest()}
		\item \texttt{kannAblegen(String farbe)}
		\item \texttt{setAusgewaehlt(boolean)}
		\item \texttt{draw(Graphics2D g)}
	\end{itemize}
\end{frame}

\begin{frame}
	\frametitle{ChemieGameObject}
	Eine Erweiterung von \texttt{BaseObject}, die das gesamte Chemie-Minigame verwaltet.

	Sie beinhaltet:
		\begin{itemize}
			\item \texttt{blockBilder}: Eine \texttt{Map<String, Image>}, die Farbnamen auf die zugehörigen Block-Bilder abbildet und an alle \texttt{Reagenzglas}-Objekte weiterreicht
			\item \texttt{glaeser}: Eine \texttt{List<Reagenzglas>}
			\item \texttt{ausgewaehltesGlas}
			\item \texttt{verbleibendeSekunden}
		\end{itemize}
\end{frame}

\begin{frame}

		\item \texttt{istGewonnen()}
		\item \texttt{update(BaseScene)}
		\item \texttt{draw(Graphics2D)}
	\end{itemize}
\end{frame}

Eine Erweiterung von \texttt{BaseObject}, die das gesamte Physik-Minigame verwaltet.

Die Klasse beinhaltet:
	\begin{itemize}
		\item \texttt{simulateButton}
		\item \texttt{basisBeschleunigungen}
		\item \texttt{kraftRichtung}
		\item \texttt{wand}
		\item \texttt{ziel}
		\item \texttt{levelInitialisieren()}
	\end{itemize}
\end{frame}
